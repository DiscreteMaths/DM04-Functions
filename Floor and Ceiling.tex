\frametitle{Floor and Ceiling Functions}

{
\large
\begin{itemize}
\item $\lceil x\rceil$ : Ceiling function
\item $\lfloor x\rfloor$  : Floor Function
\item $\{x\}$ : Fractional Part of a number
\[\{x\} = x- \lfloor x\rfloor \]
\end{itemize}
}
\end{frame}

\end{document}
